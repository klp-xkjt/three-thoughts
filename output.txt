use clap::{Parser, Subcommand};
use ts3::core::memory::Memory;

#[derive(Parser)]
#[command(version)]
struct Cli {
    #[command(subcommand)]
    command: Commands
}

#[derive(Subcommand)]
enum Commands {
    Run {
        file: String
    }
}

fn main() {
    let cli = Cli::parse();
    match cli.command {
        Commands::Run { file } => {
            let mem = Memory::new();
            println!("Memory has created, len: {}", mem.cells.len());
            println!("File: {}", file)
        }
    }
}
